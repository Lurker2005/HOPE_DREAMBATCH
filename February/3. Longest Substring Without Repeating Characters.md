## Sliding window+set
```java
class Solution {
    public int lengthOfLongestSubstring(String s) {

        if (s.length() == 0) return 0;

        HashSet<Character> set = new HashSet<>();
        int i = 0;
        int j = 0;
        int maxLength = 0;

        while (j < s.length()) {

            while (set.contains(s.charAt(j))) {
                set.remove(s.charAt(i));
                i++;
            }

            set.add(s.charAt(j));
            maxLength = Math.max(maxLength, j - i + 1);
            j++;
        }

        return maxLength;
    }
}
```
