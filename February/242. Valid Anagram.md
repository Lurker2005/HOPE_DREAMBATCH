## HashMap Approach this approach is littlebit time complexive

```java
class Solution {
    public boolean isAnagram(String s, String t) {
     HashMap<Character,Integer>map=new HashMap<>();
     int n=s.length();
     if(n==t.length()){
        for(int i=0;i<n;i++){
            map.put(s.charAt(i),map.getOrDefault(s.charAt(i),0)+1);
            map.put(t.charAt(i),map.getOrDefault(t.charAt(i),0)-1);
        }
        for(int i=0;i<n;i++){
            if(map.get(s.charAt(i))!=0){
                return false;
            }
            }
     }
     else{
        return false;
     }  
     return true;
    }
}
```

## String Apprach A Little bit optimized

```java
class Solution {
    public boolean isAnagram(String s, String t) {
       int n=s.length();
       int n1=t.length();
       if(n!=n1){
        return false;
       }
       int[] arr=new int[26];
       for(int i=0;i<n;i++){
        arr[s.charAt(i)-'a']++;
        arr[t.charAt(i)-'a']--;
       }
       for(int i=0;i<26;i++){
        if(arr[i]!=0){
            return false;
        }
       }
       return true;

    }
}
```

```java
class Solution {
    public boolean isAnagram(String s, String t) {
       int n=s.length();
       int n1=t.length();
       if(n!=n1){
        return false;
       }
       char[] str=s.toCharArray();
       char[] tstr=t.toCharArray();
       Arrays.sort(str);
       Arrays.sort(tstr);
       for(int i=0;i<n;i++){
        if(str[i]!=tstr[i]){
            return false;
        }
       }
       return true;

    }
}
```
